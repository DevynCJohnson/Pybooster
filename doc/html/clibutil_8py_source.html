68 def list_elf_lib_funcs(libfile_path: str) -> list:
69  """Return a list of functions in a shared ELF library; Requires 'nm'"""
70  if isfile(libfile_path):
71  libobj = getoutput(r'nm -D --defined-only ' + libfile_path)
72  objs = findall('([a-f0-9]+) T (?!_)(.+)', libobj)
73  lst = []
74  for i in objs:
75  lst.append(i[1])
76  return lst
77  else:
78  raise FileNotFoundError(r'Library file (' + libfile_path + r') not found!')
