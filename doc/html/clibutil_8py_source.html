69 def list_elf_lib_funcs(libfile_path: str) -> list:
70  """Return a list of functions in a shared ELF library
71  Requires 'nm'
72  """
73  if isfile(libfile_path):
74  libobj = getoutput('nm -D --defined-only ' + libfile_path)
75  objs = findall('([a-f0-9]+) T (?!_)(.+)', libobj)
76  lst = []
77  for i in objs:
