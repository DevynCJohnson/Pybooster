93 def getvalue(m1: float, m2: float, hue: float) -> float:
94  """Get value from hue"""
95  hue = hue % 1.0

314 def hsv2rgb(h: float, s: float, v: float) -> tuple:
315  """HSV -> RGB"""
316  if s == 0.0:
317  return v, v, v
318  i = int(h * 6.0)
319  f = (h * 6.0) - i
320  p = v * (1.0 - s)
321  q = v * (1.0 - s * f)
322  t = v * (1.0 - s * (1.0 - f))
323  i = i % 6
324  if i == 0:
325  return v, t, p
326  elif i == 1:
327  return q, v, p
