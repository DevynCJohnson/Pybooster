52 /** Private Function: Returns the Square-Root of a positive number (Limit: 1000000000000000.0) */
53 static double squareroot(const double num) {
54  if (num < (double)0.0) { return -1.0; }
55  else if (floor(num) == num && num <= (double)81.0) {
56  if (num == (double)0.0) return 0.0;
57  else if (num == (double)1.0) return 1.0;
58  else if (num == (double)4.0) return 2.0;
59  else if (num == (double)9.0) return 3.0;
60  else if (num == (double)16.0) return 4.0;
61  else if (num == (double)25.0) return 5.0;
62  else if (num == (double)36.0) return 6.0;
63  else if (num == (double)49.0) return 7.0;
64  else if (num == (double)64.0) return 8.0;
65  else if (num == (double)81.0) return 9.0;
66  }
67  register double start = (double)0.0; // if num < 1; then 0 < num < sqrt(num); else 0 < sqrt(num) < num
68  register double end = (num < (double)1.0 ? (double)1.0 : num);
69  register double mid, midSqr;
70  while ((end - start) > (double)0.00001) { // Define precision
71  mid = (start + end) * (double)0.5;
72  midSqr = mid * mid;
73  if (midSqr == num) return mid; // Exact sqrt value
74  else if (midSqr < num) start = mid;
75  else end = mid;
76  }
77  return ((start + end) * (double)0.5);
78 }

152 doublepair_t xy2polar(const double x, const double y) {
153  doublepair_t polar;
154  polar.y = (1.0 / tan(y / x)); // tan-1(y / x); angle
155  polar.x = squareroot((x * x) + (y * y)); // hypot
156  return polar;
