184 /**
185  Calculate the pKa from Ka (Acid Dissociation Constant)
186 
187  Chemistry Explanation\n
188  [HA] = [H+] + [A-]; Molar concentrations\n
189  Ka = ([H+][A-]) / [HA]\n
190  Ka = ([X][0.01M+X]) / (0.02M-X); Ka = 10^-5M\n
191  X = 0.00001 * (0.02 / 0.01) = 2*(10^-5) = [H+]\n
192  pH = -log(2*(10^-5)) = 4.7
193 
194  @param[in] ka Acid Dissociation Constant
195  @returns pka value
196 */
197 double pka(const double ka) {
198  return (0.0 - log10(ka));
199 }
