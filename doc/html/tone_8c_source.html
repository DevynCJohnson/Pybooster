84 /**
85  Generate a tone lasting one second; remember to use `free(sound.buffer);`
86 
87  @param[in] freq Frequency of sound wave
88  @param[in] format Sound format in the ao_sample format
89  @returns sound (tone datatype)
90 */
91 tone generatetone(const float freq, ao_sample_format format) {
92  tone sound;
93  sound.buf_size = (format.bits >> 3) * format.channels * format.rate;
94  register int i;
95  register float _val;
96  for (i = 0; i < format.rate; ++i) {
97  _val = (((float)i / (float)format.rate) * (float)6.283185F);
98  sound.sample = (int)((float)24576.0F * sinf(freq * _val)); // amplitude * sin_wave
99  sound.buffer[4 * i] = sound.buffer[4 * i + 2] = (char)(sound.sample & 0xff);
100  sound.buffer[4 * i + 1] = sound.buffer[4 * i + 3] = (char)((sound.sample >> 0x08) & 0xff);
101  }
102  return sound;
103 }
