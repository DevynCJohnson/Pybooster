103 def findgw() -> str:
104  """Get the Gateway IP address"""
105  with open(r'/proc/net/route', mode='rt', encoding='utf-8') as _file:
106  for _line in _file:
107  _field = _line.strip().split()
108  if _field[1] != '00000000' or not int(_field[3], 16) & 2:
109  continue
