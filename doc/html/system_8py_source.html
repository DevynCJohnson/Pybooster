220 def which(program: str) -> str:
221  """Return the path of the specified application (if it exists)
222  An empty string is returned if the program does not exist
223  """
224  fpath = path.split(program)[0]
225  if fpath and path.isfile(program) and access(program, X_OK):
226  return program
227  else:
228  envpath = environ['PATH'].split(pathsep)
229  for filepath in envpath:
230  exe_file = path.join(filepath.strip('"'), program)

232  return exe_file
233  return ''
