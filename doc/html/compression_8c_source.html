62 /**
63  zlib stream compression
64 
65  @param[in] source File to compress
66  @param[in] dest File destination for compressed data
67  @param[in] level Compression level (0-9)
68  @returns Return value
69 */
70 int zlib_compress_stream(FILE *__restrict__ source, FILE *__restrict__ dest, const int level) {
71  int ret, flush;
72  unsigned have;
73  z_stream strm;
74  unsigned char in[CHUNK];
75  unsigned char out[CHUNK];
76  // Allocate deflate state
77  strm.zalloc = Z_NULL;
78  strm.zfree = Z_NULL;
79  strm.opaque = Z_NULL;
80  ret = deflateInit(&strm, level);
81  if (ret != Z_OK) return ret;
82  do { // Compress until end of file
83  strm.avail_in = (unsigned int)fread(in, 1, CHUNK, source);
84  if (ferror(source)) {
85  (void)deflateEnd(&strm);
86  return Z_ERRNO;
87  }
88  flush = (feof(source) ? Z_FINISH : Z_NO_FLUSH);
89  strm.next_in = in;
90  do { // Run deflate() on input until output buffer is not full
91  strm.avail_out = CHUNK;
92  strm.next_out = out;
93  ret = deflate(&strm, flush); // No bad return value
94  if (ret == Z_STREAM_ERROR) return Z_STREAM_ERROR; // State not clobbered
95  have = (CHUNK - strm.avail_out);
96  if ((fwrite(out, sizeof(char), have, dest)) != have || ferror(dest)) {
97  (void)deflateEnd(&strm);
98  return Z_ERRNO;
99  }
100  } while (strm.avail_out == 0); // Finish compression if all data has been read
101  if (strm.avail_in != 0x00) return Z_STREAM_ERROR; // All input will be used
102  // Done when all data has been processed
103  } while (flush != Z_FINISH);
104  if (ret != Z_STREAM_END) return Z_STREAM_ERROR; // Stream is complete
105  // Clean up and return
106  (void)deflateEnd(&strm);
107  return Z_OK;
108 }

120  FILE *source = fopen(source_filename, "rb");
121  if (source == (FILE *)NULL) {
122  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
123  exit(EXIT_FAILURE);
124  }
125  FILE *dest = fopen(dest_filename, "wb");
126  if (dest == (FILE *)NULL) {
127  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
128  fclose(source);
129  exit(EXIT_FAILURE);
130  }
131  // Prepare source file and buffer
132  ulint source_size = getfilesize(source_filename);
133  Bytef *source_data = (Bytef *)calloc(source_size + 1, 0x01);
134  if (source_data == (Bytef *)NULL) { ALLOC_ERROR() }
135  if (fread(source_data, 0x01, source_size, source) != source_size) {
136  free(source_data);
137  UNABLE_TO_READ_FILE()
138  }
139  fclose(source);
140  source_data[source_size + 1] = '\0';
141  // Prepare dest file
142  ulint dest_size = compressBound(source_size + 1024);
143  Bytef *dest_buf = (Bytef *)calloc(dest_size, 0x01);
144  if (dest_buf == (Bytef *)NULL) {
145  fclose(dest);
146  free(source_data);
147  free(dest_buf);
148  ALLOC_WRITE_ERROR()
149  }
150  // Compress data
151  int ret;
152  ret = compress2(dest_buf, &dest_size, source_data, source_size, level);
153  if (ret != Z_OK) {

155  free(dest_buf);
156  return ret;
157  }
158  // Write compressed data
159  setbuf(dest, (char *)dest_buf);
160  if ((fwrite((char *)dest_buf, 0x01, dest_size, dest)) != dest_size) {
161  fclose(dest);
162  free(dest_buf);
163  UNABLE_TO_WRITE_FILE()
164  }
165  // Clean up and return
166  fclose(dest);
167  free(dest_buf);

179 int zlib_decompress_stream(FILE *__restrict__ source, FILE *__restrict__ dest) {
180  int ret;
181  unsigned have;
182  z_stream strm;
183  unsigned char in[CHUNK];
184  unsigned char out[CHUNK];
185  // Allocate inflate state
186  strm.zalloc = Z_NULL;
187  strm.zfree = Z_NULL;
188  strm.opaque = Z_NULL;
189  strm.avail_in = 0;
190  strm.next_in = Z_NULL;
191  ret = inflateInit(&strm);
192  if (ret != Z_OK) return ret;
193  do { // Decompress until deflate stream ends or end of file
194  strm.avail_in = (unsigned int)fread(in, 1, CHUNK, source);
195  if (ferror(source)) {
196  (void)inflateEnd(&strm);
197  return Z_ERRNO;
198  }
199  if (strm.avail_in == 0x00) break;
200  strm.next_in = in;
201  do { // Run inflate() on data until the output buffer is not full
202  strm.avail_out = CHUNK;
203  strm.next_out = out;
204  ret = inflate(&strm, Z_NO_FLUSH);
205  if (ret == Z_STREAM_ERROR) return Z_STREAM_ERROR; // State not clobbered
206  switch (ret) {
207  case Z_NEED_DICT:
208  ret = Z_DATA_ERROR; // Fall through
209  case Z_DATA_ERROR:
210  case Z_MEM_ERROR:
211  (void)inflateEnd(&strm);
212  return ret;
213  default: continue;
214  }
215  have = (CHUNK - strm.avail_out);
216  if ((fwrite(out, sizeof(char), have, dest)) != have || ferror(dest)) {
217  (void)inflateEnd(&strm);
218  return Z_ERRNO;
219  }
220  } while (strm.avail_out == 0);
221  } while (ret != Z_STREAM_END); // Done when inflate() finishes
222  // Clean up and return
223  (void)inflateEnd(&strm);
224  return ((ret == Z_STREAM_END) ? Z_OK : Z_DATA_ERROR);
225 }

235 int zlib_decompress_file(char *source_filename, char *dest_filename) {
236  FILE *source = fopen(source_filename, "rb");
237  if (source == NULL) {
238  fprintf(stderr, "ERROR: Failed to open source file!\n%s\n", strerror(errno));
239  exit(EXIT_FAILURE);
240  }
241  FILE *dest = fopen(dest_filename, "wb");
242  if (dest == NULL) {
243  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
244  fclose(source);
245  exit(EXIT_FAILURE);
246  }
247  // Prepare source file and buffer
248  ulint source_size = getfilesize(source_filename);
249  uint8_t *source_data = (uint8_t *)calloc(source_size + 1, 0x01);
250  if (source_data == NULL) { ALLOC_ERROR() }
251  if (fread(source_data, 0x01, source_size + 1, source) != source_size) {
252  fputs("`zlib_decompress_file` (Reading source file): ", stderr);
253  free(source_data);
254  fclose(source);
255  fclose(dest);
256  UNABLE_TO_READ_FILE()
257  }
258  fclose(source);
259  source_data[source_size + 1] = '\0';
260  // Allocate inflate state
261  unsigned char output_data[CHUNK] = { 0 };
262  z_stream strm;
263  strm.zalloc = Z_NULL;
264  strm.zfree = Z_NULL;
265  strm.opaque = Z_NULL;
266  strm.avail_in = (unsigned int)source_size;
267  strm.next_in = (Bytef *)source_data;
268  strm.data_type = Z_BINARY;
269  int ret = inflateInit(&strm);
270  if (ret != Z_OK) {
271  free(source_data);
272  fclose(dest);
273  return ret;
274  }
275  size_t dest_size = 0x00;
276  // Decompress data
277  do { // Run inflate() on data until finished
278  if (strm.avail_in == 0) break;
279  strm.avail_out = CHUNK;
280  strm.next_out = (Bytef *)output_data;
281  ret = inflate(&strm, Z_NO_FLUSH);
282  if (ret == Z_STREAM_ERROR) { // State not clobbered
283  free(source_data);
284  fclose(dest);
285  return Z_STREAM_ERROR;
286  }
287  switch (ret) {
288  case Z_NEED_DICT:
289  ret = Z_DATA_ERROR; // Fall through
290  case Z_DATA_ERROR:
291  case Z_MEM_ERROR:
292  (void)inflateEnd(&strm);
293  free(source_data);
294  fclose(dest);
295  return ret;
296  default:
297  dest_size += (size_t)(CHUNK - strm.avail_out);
298  if (strm.avail_in == 0 || strm.avail_out == 0 || ret == Z_STREAM_END) break;
299  }
300  } while (ret != Z_STREAM_END && strm.avail_out == 0); // Wait until inflate() finishes
301  (void)inflateEnd(&strm);
302  Bytef *dest_buf = (Bytef *)calloc(dest_size, 0x01);
303  ret = uncompress(dest_buf, &dest_size, source_data, source_size);
304  if (ret != Z_OK) {
305  fclose(dest);
306  free(dest_buf);
307  free(source_data);
308  return ret;
309  }
310  free(source_data);
311  // Write compressed data
312  setbuf(dest, NULL);
313  if ((fwrite((char *)dest_buf, 0x01, dest_size, dest)) != dest_size) {
314  fclose(dest);
315  free(dest_buf);
316  fputs("`zlib_decompress_file` (Writing compressed file): ", stderr);
317  UNABLE_TO_WRITE_FILE()
318  }
319  // Clean up and return
320  fclose(dest);
321  free(dest_buf);
322  return Z_OK;
323 }
