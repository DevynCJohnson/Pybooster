64 /** Private Function: Returns the Square-Root of a positive number (Limit: 1000000000000000.0) */
65 static double squareroot(const double num) {
66  if (num < (double)0.0) { return -1.0; }
67  else if (floor(num) == num && num <= (double)81.0) {
68  if (num == (double)0.0) return 0.0;
69  else if (num == (double)1.0) return 1.0;
70  else if (num == (double)4.0) return 2.0;
71  else if (num == (double)9.0) return 3.0;
72  else if (num == (double)16.0) return 4.0;
73  else if (num == (double)25.0) return 5.0;
74  else if (num == (double)36.0) return 6.0;
75  else if (num == (double)49.0) return 7.0;
76  else if (num == (double)64.0) return 8.0;
77  else if (num == (double)81.0) return 9.0;
78  }
79  register double start = (double)0.0; // if num < 1; then 0 < num < sqrt(num); else 0 < sqrt(num) < num
80  register double end = (num < (double)1.0 ? (double)1.0 : num);
81  register double mid, midSqr;
82  while ((end - start) > (double)PY_DOUBLE_PRECISION) { // Define precision
83  mid = ((start + end) * (double)0.5);
84  midSqr = (mid * mid);
85  if (midSqr == num) { return mid; } // Exact sqrt value
86  else if (midSqr < num) { start = mid; }
87  else { end = mid; }
88  }
89  return ((start + end) * (double)0.5);
90 }
