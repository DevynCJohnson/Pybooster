89 class NeuroCode:
90  """Neurocode class"""
91  def __init__(self, data: list, layers: list, iterations: int=40000, rate: float=0.4):
92  """Initiate Neurocode-learning
93 
94  - layers: Hidden neural layers (neurocodes between the input and output layers); Specify the number of hidden layers in the network and the size of each layer. For example, `layers = [3, 4]` makes two hidden layers, the first with 3 nodes and the second with 4 nodes. By default, one hidden layer is used with a size proportionate to the size of the input array.
95  - error_thresh: Error threshold goal (float less than 1.0)
96  - rate: Learning rate (float less than 1.0)
97  """
98  input_size = len(data[0][0])
99  output_size = len(data[0][1])
100  if not layers:
101  self.hidden_layers = [max(3, math.floor(input_size / 2))]
102  else:
103  self.hidden_layers = layers
104  self.sizes = list(flatten([input_size, self.hidden_layers, output_size]))
105  self.iterations = iterations
106  self.rate = rate
107  self.io_rules = data
108  self.io_rules_len = len(data)
109  self.outputlayer = len(self.sizes) - 1
110  # Training State
111  self.deltas = [[]] * (self.outputlayer + 1)
112  self.changes = [[]] * (self.outputlayer + 1)
113  self.errors = [[]] * (self.outputlayer + 1)
114  self.outputs = [[]] * (self.outputlayer + 1)
115  self.biases = [[]] * (self.outputlayer + 1)
116  self.weights = [[]] * (self.outputlayer + 1)
117  for layer in range(self.outputlayer + 1):
118  _size = self.sizes[layer]
119  self.deltas[layer] = [0] * _size
120  self.errors[layer] = [0] * _size
121  self.outputs[layer] = [0] * _size
122  if layer > 0:
123  self.biases[layer] = [(random() * 0.4) - 0.2 for i in range(_size)]
124  self.weights[layer] = [0] * _size
125  self.changes[layer] = self.weights[layer]
126  for node in range(_size):
127  _prev_size = self.sizes[layer - 1]
128  self.weights[layer][node] = [(random() * 0.4) - 0.2 for j in range(_prev_size)]
129  self.changes[layer][node] = [0] * _prev_size

131  def train(self) -> tuple: # noqa C901
132  """Main Neurocode training"""
133  error = 1.0
134  used_iterations = 0
135  for i in range(self.iterations):
136  used_iterations = i
137  if error <= 0.0001: # Error Threshold
138  break
139  _sum = 0.0
140  for d in self.io_rules:
141  self.run(d[0])
142  self._calculate_deltas(d[1])
143  # Adjust Weights
144  for layer in range(1, self.outputlayer + 1):
145  incoming = self.outputs[layer - 1]
146  for node in range(self.sizes[layer]):
147  delta = self.deltas[layer][node]
148  for k in range(len(incoming)):
149  change = self.changes[layer][node][k]
150  change = (self.rate * delta * incoming[k]) + (0.1 * change) # 0.1 = momentum
151  self.changes[layer][node][k] = change
152  self.weights[layer][node][k] = change + self.weights[layer][node][k]
153  self.biases[layer][node] = self.biases[layer][node] + (self.rate * delta)
154  _errsum = 0.0
155  for err in self.errors[self.outputlayer]:
156  _errsum = _errsum + (err ** 2.0)
157  err = _errsum / len(self.errors[self.outputlayer])
158  _sum = _sum + err
159  error = _sum / self.io_rules_len
160  return (error, used_iterations) # (float, int)
161 
162  def run(self, _input) -> list:
163  """Forward Propagation; Execute neuralnet"""
164  output = self.outputs[0] = _input # Set output state of input layer
165  for layer in range(1, self.outputlayer + 1):
166  for node in range(self.sizes[layer]):
167  weights = self.weights[layer][node]
168  _sum = self.biases[layer][node]
169  for k in range(len(weights)):
170  _sum = _sum + (weights[k] * _input[k])
171  self.outputs[layer][node] = 1 / (1 + math.exp(-_sum))
172  _input = self.outputs[layer]
173  output = _input
174  return output

176  def _calculate_deltas(self, target: list):
177  """Backward Propagation"""
178  layer = self.outputlayer
179  while layer >= 0:
180  for node in range(self.sizes[layer]):
181  output = self.outputs[layer][node]
182  error = 0.0
183  if layer == self.outputlayer:
184  error = target[node] - output
185  else:
186  deltas = self.deltas[layer + 1]
187  for k in range(len(deltas)):
188  error = error + (deltas[k] * self.weights[layer + 1][k][node])
189  self.errors[layer][node] = error
190  self.deltas[layer][node] = (error * output) * (1 - output)
191  layer -= 1

283  def to_c_function(self, fnname: str='nn_run', indent: int=0):
284  """Convert the neural-network to C code"""
285  fn = 'void {fnname}(double *i, double *o)'.format(fnname=fnname) + '{\n'
286  terms = {}
287  oterms = {}
288  lterms = []
289  for k in range(self.sizes[0]):
290  lterms.append(r'o0_' + str(k))
291  terms[lterms[-1]] = r'i[' + str(k) + r']'
292  for l in range(1, self.outputlayer + 1):
293  size = self.sizes[l]
294  for n in range(size):
295  term = str(-self.biases[l][n])
296  length = len(self.weights[l][n])
297  for k in range(length):
298  w = self.weights[l][n][k]
299  term = term + (r'-' if w > 0 else r'+') + str(abs(w)) + r'*o' + str(l - 1) + r'_' + str(k)
300  v = r'(1/(1+exp(' + term + r')))'
301  for k in lterms:
302  v = v.replace(k, terms[k])
303  lterms.append(r'o' + str(l) + r'_' + str(n))
304  terms[lterms[-1]] = v
305  if l == self.outputlayer:
306  oterms[r'o' + str(l) + r'_' + str(n)] = r'o[' + str(n) + r']'
307  for k, v in oterms.items():
308  fn += r' ' + v + r' = ' + terms[k] + ';\n'
309  fn += '}\n'
310  return _indent(fn, indent)
