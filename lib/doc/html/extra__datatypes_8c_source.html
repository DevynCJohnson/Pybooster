106 /**
107  Push an element into stack; return NULL on error\n
108  Usage: head = push_linked_stack(head, element);
109 */
110 linked_stack_node_t *push_linked_stack(linked_stack_node_t *head, const uint64_t data) {
111  linked_stack_node_t *tmp = (linked_stack_node_t *)calloc(0x01, sizeof(linked_stack_node_t));
112  if (tmp == (linked_stack_node_t *)NULL) { free(tmp); return NULL; }
113  tmp->data = data;
114  tmp->next = head;
115  head = tmp;
116  return head;
117 }

120 /** Pop an element from the stack; Usage: head = pop_linked_stack(head, &element); */
121 linked_stack_node_t *pop_linked_stack(linked_stack_node_t *head, uint64_t *element) {
122  linked_stack_node_t *tmp = head;
123  *element = head->data;
124  head = head->next;
125  free(tmp);
126  return head;
127 }

183  Binary Heap Datatype\n
184  Initialze a binary heap\n
185  Usage: PriorityQueue H; init_heap(MAXSIZE);
186 */
187 PriorityQueue init_heap(size_t MaxElements) {
188  PriorityQueue H;
189  if (MaxElements < 10) { return NULL; } // Is MaxElements less than the minimum priority queue
190  H = calloc(0x1, sizeof(binary_heap_t));
191  if (H == NULL) { return NULL; }
192  H->Elements = calloc((size_t)(MaxElements + 0x1), sizeof(ElementType));
193  if (H->Elements == NULL) { return NULL; }
194  H->Capacity = MaxElements;
195  H->Size = 0x0;
196  H->Elements[0] = -32767; // H->Element[0] is a sentinel
197  return H;
198 }

205 void heapinsert(ElementType X, PriorityQueue H) {
206  if (!isheapfull(H)) {
207  register size_t i;
208  for (i = ++H->Size; H->Elements[i / 0x2] > X; i /= 0x2) {
209  H->Elements[i] = H->Elements[i / 0x2];

216 /** Remove smallest element from binary-heap */
217 ElementType deletemin_heap(PriorityQueue H) {
218  ElementType MinElement, LastElement;
219  if (isheapempty(H)) { return H->Elements[0x0]; }
220  MinElement = H->Elements[0x1];
221  LastElement = H->Elements[H->Size--];
222  register size_t i, Child;
223  for (i = 0x1; (i * 0x2) <= H->Size; i = Child) { // Find smaller child
224  Child = (i * 0x2);
225  if (Child != (H->Size) && ((H->Elements[Child + 0x1]) < (H->Elements[Child])))
226  Child++;
227  // Percolate one level
228  if (LastElement > (H->Elements[Child])) H->Elements[i] = H->Elements[Child];
229  else break;
230  }
231  H->Elements[i] = LastElement;
232  return MinElement;
233 }
